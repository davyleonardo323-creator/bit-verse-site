C++

C++ é uma linguagem rápida e poderosa, muito usada em jogos, sistemas e software de alta performance.


#include <iostream>
using namespace std;

void fibonacci(int n) {
  int a = 0, b = 1;
  for (int i = 0; i < n; i++) {
    cout << a << endl;
    int next = a + b;
    a = b;
    b = next;
  }
}

int main() {
  fibonacci(10);
  return 0;
}

Rust

Rust é uma linguagem segura e rápida, focada em performance e segurança sem garbage collector.


fn fibonacci(n: u32) {
    let (mut a, mut b) = (0, 1);
    for _ in 0..n {
        println!("{}", a);
        let next = a + b;
        a = b;
        b = next;
    }
}

fn main() {
    fibonacci(10);
}
